Examples

    align(names1=x1, names2=x2)
    # [1] 1 3 2 4
    
    x1 <- LETTERS[1:4]
    x2 <- c(1,3,2,4)
     align(names1=x1, names2=x2)
    # [1] 1 3 2 4

    # [1] 1 3 2 4
